**해결책:**
```typescript
// 작업 완료 후에도 문서 목록을 계속 조회
if (status === 'completed') {
  // 폴링은 중지하되, 문서 목록은 계속 조회
  // 문서 생성이 완료될 때까지 조회 (예: 30분)
  const documentPollingInterval = setInterval(async () => {
    const result = await refetchDocuments();
    const docs = result.data?.data?.documents || [];
    if (docs.length >= expectedCount) {
      clearInterval(documentPollingInterval);
    }
  }, 5000); // 5초마다 조회
  
  // 30분 후 자동 중지
  setTimeout(() => {
    clearInterval(documentPollingInterval);
  }, 30 * 60 * 1000);
}
```
